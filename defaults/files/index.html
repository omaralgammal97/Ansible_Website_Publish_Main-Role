Ansible Task by Omar Ali Al-Gammal

Description: This task installs Apache, enables the service, backs up its configuration, changes the listening port, and restarts the server and publish website.

            ---
            - name: Apache Setup and Port Change (by Omar Ali Al-Gammal)
              hosts: nti
              become: yes
            
              tasks:
                - name: Install Apache (httpd)
                  dnf:
                    name: httpd
                    state: present
            
                - name: Enable and start Apache
                  service:
                    name: httpd
                    state: started
                    enabled: yes
            
                - name: Backup httpd.conf
                  copy:
                    src: /etc/httpd/conf/httpd.conf
                    dest: /etc/httpd/conf/httpd.conf.bak
                    remote_src: yes
            
                - name: Change Apache listen port to 88
                  lineinfile:
                    path: /etc/httpd/conf/httpd.conf
                    regexp: '^Listen '
                    line: 'Listen 88'
            
                - name: Allow Apache to use port 88 in SELinux
                  command: semanage port -m -t http_port_t -p tcp 88
                  args:
                    warn: false
                  ignore_errors: true
            
                - name: Ensure firewalld allows port 88
                  firewalld:
                    port: 88/tcp
                    permanent: yes
                    state: enabled
                    immediate: yes
            
                - name: Restart Apache
                  service:
                    name: httpd
                    state: restarted
                    
                - name: task4 upload page
                  copy:
                    src: ./index.html
                    dest: /var/www/html/index.html
